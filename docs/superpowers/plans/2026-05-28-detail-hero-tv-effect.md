# Detail Hero Old-TV Effect Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a continuous animated "old-TV / VHS" effect (Ken Burns drift + dense static + chroma bleed + tracking band + scanlines + vignette) behind the shared detail-page hero on all four detail pages, with a hover-reveal corner toggle backed by a persisted global preference.

**Architecture:** A self-contained presentational overlay component (`DetailHeroEffect`) is layered into the existing shared `DetailHero` between the (now-drifting) background and the existing scrim. A tiny external-store module (`heroTvEffect.ts`, using React 19's `useSyncExternalStore`) holds the global on/off boolean, loads it from the existing `tauri-plugin-store` singleton, and persists changes — so `DetailHero` reads it via a hook and no consumer components or App.tsx startup code need to change. Effect is GPU-only (CSS transforms + tiled PNG noise), pauses when off-screen/hidden, and falls back to a static treatment under `prefers-reduced-motion`.

**Tech Stack:** React 19 + TypeScript, Vite, CSS (skin custom properties), `@tauri-apps/plugin-store` (via `src/store.ts`), Vitest + @testing-library/react (jsdom).

## Design Notes (read before starting)

**Why an external-store module instead of prop-drilling (deviation from spec wording):**
The spec said the boolean would be "threaded to `DetailHero` the same way other UI prefs reach shared components." Investigation showed that the four detail views (`ArtistDetailContent`, `AlbumDetail`, `TagDetail`, `TrackDetailView`) receive **only entity-identifying props** — no app-level UI state is drilled to them — and App.tsx's startup restore is a fragile positional-tuple destructure (`src/App.tsx:1629`). Threading a new prop would mean editing that tuple plus four component interfaces plus `DetailHero`'s props. Instead we use a small external store + hook (the `useSyncExternalStore` pattern already used in `src/videoFrameQueue.ts`, `src/components/HomeHero.tsx`, `src/components/HomeShelf.tsx`). This keeps the persistence in the same `store` singleton (`src/store.ts`, `autoSave: 500` debounce) and default-on semantics, while touching zero consumers. This is an implementation decision within the approved scope (global toggle, persisted, default on), not a scope change.

**Z-order inside `.detail-hero` after this change:**
```
.detail-hero-bg       z-index 0   (montage, now drifting via Ken Burns)
.detail-hero-effect   z-index 1   (FX overlay — full strength)
.detail-hero::after   z-index 2   (existing scrim — was 0)
.detail-hero-row      z-index 3   (art + title + actions — was 1)
.detail-hero-tv-toggle z-index 4  (corner checkbox)
```
The effect sits **below** the scrim so the existing scrim gradient still provides baseline legibility, while snow plays full-strength across the whole hero (per the approved "full strength everywhere" decision).

Run this exact command (Python 3, no external deps):

```bash
python3 - <<'PY'
import zlib, struct, random
random.seed(1729)  # deterministic, reproducible asset
size = 128
raw = bytearray()
for y in range(size):
    raw.append(0)  # PNG filter type 0 per scanline
    for x in range(size):
        v = random.randint(0, 255)   # luminance
        a = random.randint(0, 255)   # vary alpha so grain sparkles
        raw += bytes([v, a])
def chunk(typ, data):
    c = typ + data
    return struct.pack(">I", len(data)) + c + struct.pack(">I", zlib.crc32(c) & 0xffffffff)
sig = b'\x89PNG\r\n\x1a\n'
ihdr = struct.pack(">IIBBBBB", size, size, 8, 4, 0, 0, 0)  # 8-bit grayscale+alpha
idat = zlib.compress(bytes(raw), 9)
png = sig + chunk(b'IHDR', ihdr) + chunk(b'IDAT', idat) + chunk(b'IEND', b'')
open('src/assets/tv-noise.png', 'wb').write(png)
print("wrote src/assets/tv-noise.png", len(png), "bytes")
PY
```

Create `src/heroTvEffect.ts`:

```ts
import { useSyncExternalStore } from "react";
import { store } from "./store";

const KEY = "heroTvEffect";

let enabled = true;
let loadStarted = false;
const listeners = new Set<() => void>();

function emit() {
  for (const l of listeners) l();
}

async function load(): Promise<void> {
  try {
    const v = await store.get<boolean>(KEY);
    if (typeof v === "boolean" && v !== enabled) {
      enabled = v;
      emit();
    }
  } catch (e) {
    console.error("Failed to load heroTvEffect preference:", e);
  }
}

export function getHeroTvEffectSnapshot(): boolean {
  return enabled;
}

export function setHeroTvEffect(value: boolean): void {
  if (value === enabled) return;
  enabled = value;
  emit();
  store
    .set(KEY, value)
    .catch((e) => console.error("Failed to persist heroTvEffect preference:", e));
}

export function subscribeHeroTvEffect(cb: () => void): () => void {
  listeners.add(cb);
  // Lazily load the persisted value the first time anyone subscribes (i.e. the
  // first detail hero mounts) — by then the app is well past Tauri startup.
  if (!loadStarted) {
    loadStarted = true;
    void load();
  }
  return () => {
    listeners.delete(cb);
  };
}

export function useHeroTvEffect(): [boolean, (v: boolean) => void] {
  const value = useSyncExternalStore(
    subscribeHeroTvEffect,
    getHeroTvEffectSnapshot,
    getHeroTvEffectSnapshot,
  );
  return [value, setHeroTvEffect];
}

// --- test-only helpers ---
export function loadHeroTvEffectForTest(): Promise<void> {
  return load();
}
export function __resetHeroTvEffectForTest(): void {
  enabled = true;
  loadStarted = false;
  listeners.clear();
}
```

Create `src/components/DetailHeroEffect.tsx`:

```tsx
import { useEffect, useRef } from "react";
import tvNoise from "../assets/tv-noise.png";
import "./DetailHeroEffect.css";

interface Props {
  /** When false, renders nothing (effect fully torn down). */
  active: boolean;
}

/**
 * Old-TV / VHS background effect for the detail hero. Pure presentational +
 * GPU-only CSS animation. Pauses (via the `tv-paused` class) when the hero is
 * scrolled out of view or the window is hidden, so it never burns cycles in the
 * background. Static fallback for prefers-reduced-motion lives in the CSS.
 */
export function DetailHeroEffect({ active }: Props) {
  const rootRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    if (!active) return;
    const el = rootRef.current;
    if (!el) return;

    let onScreen = true;
    let pageVisible = !document.hidden;

    const apply = () => {
      el.classList.toggle("tv-paused", !(onScreen && pageVisible));
    };

    const io =
      typeof IntersectionObserver !== "undefined"
        ? new IntersectionObserver(
            (entries) => {
              onScreen = entries[0]?.isIntersecting ?? true;
              apply();
            },
            { threshold: 0 },
          )
        : null;
    io?.observe(el);

    const onVisibility = () => {
      pageVisible = !document.hidden;
      apply();
    };
    document.addEventListener("visibilitychange", onVisibility);
    apply();

    return () => {
      io?.disconnect();
      document.removeEventListener("visibilitychange", onVisibility);
    };
  }, [active]);

  if (!active) return null;

  return (
    <div
      ref={rootRef}
      className="detail-hero-effect"
      aria-hidden="true"
      style={{ ["--tv-noise" as string]: `url(${tvNoise})` }}
    >
      <div className="tv-bleed" />
      <div className="tv-bleed-2" />
      <div className="tv-scan" />
      <div className="tv-track" />
      <div className="tv-noise" />
      <div className="tv-noise-2" />
      <div className="tv-vignette" />
    </div>
  );
}
```
